#include <stdio.h>
int main() {
    int x=65;  //declaring variable x as integer type and initializing to 65
    printf("The value of x is %d\n",x); // %d is a format specififier
    // 65 in binary(2進数) is 01000001
    return 0;
}

compile
"The value of x is 65"


#include <stdio.h>
int main() {
    int x=65;  //declaring variable x as integer type and initializing to 65
    printf("The value of x is %c\n",x); // %c is a character number?
    // 65 in binary is 01000001
    return 0;
}

compile
"The value of x is A" A→from Ascii table


#include <stdio.h>
int main() {
    int x=65;  //declaring variable x as integer type and initializing to 65
    printf("The value of x is %o\n",x); // %o is a octonumber 8進数
    // 65 in binary is 01000001
    return 0;
}

compile
"The value of x is 101" A→from Ascii table


#include <stdio.h>
int main() {
    char x='A';  //declaring variable x as character type and initializing to A
    printf("The value of x is %c\n",x); // %c is a format specififier
    // 65 in binary(2進数) is 01000001
    return 0;
}

compile
"The value of x is A"
